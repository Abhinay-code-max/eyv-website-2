async def generate_single_plan(preferences: Dict, plan_type: str, trip_id: str, user_id: str) -> Dict:
    """Generate a single vacation plan using AI"""
    
    currency = preferences.get('currency', 'INR')
    currency_symbol = '₹' if currency == 'INR' else '$'
    
    # Tier-specific instructions — each tier now gets its OWN logic, not just Budget rules for everyone
    if plan_type == "Budget":
        tier_instructions = """
**BUDGET TIER RULES (CRITICAL):**
- Prioritize the MINIMUM possible cost for every recommendation
- Suggest budget hostels, dormitories, homestays, or guesthouses (₹500-₹1500/night)
- Recommend public transport: state buses, sleeper trains (3AC/SL), local metro, shared autos
- Avoid private taxis unless absolutely necessary; prefer shared rides
- Suggest free attractions (temples, beaches, parks, viewpoints) and skip expensive entry fees where possible
- Recommend local street food and dhabas for meals (₹100-₹250 per meal)
- For flights: only when train/bus would take 24+ hours, otherwise prefer rail
- Cap daily expenditure to absolute minimum - aim for ₹2000-₹3500 per person/day
- Clearly mark free activities with cost: 0
"""
    elif plan_type == "Premium":
        tier_instructions = """
**PREMIUM TIER RULES (CRITICAL):**
- Mid-range to upscale 3-4 star hotels with good reviews (₹3500-₹7000/night)
- Mix of flights for long distances and comfortable train classes (2AC/1AC) for shorter routes
- Private cabs/cars for local transport, occasional ride-shares
- Good restaurants with a few notable/popular dining spots, avoid only street food
- Balanced mix of free and paid activities, prioritize well-rated paid experiences
- Aim for ₹6000-₹10000 per person/day
"""
    else:  # Luxury
        tier_instructions = """
**LUXURY TIER RULES (CRITICAL):**
- 5-star hotels, resorts, or boutique luxury stays (₹12000+/night)
- Flights in business/premium class for any flight segment; private transfers for ground travel
- Fine dining and exclusive/signature restaurants each day
- Premium and exclusive experiences: private guides, skip-the-line access, concierge-level service
- Do not optimize for cost — optimize for quality, comfort, and experience
- Aim for ₹15000+ per person/day (cost is not the constraint)
"""
    
    prompt = f"""You are an expert Indian travel planner. Create a detailed {plan_type} vacation plan based on these preferences:

Destination: {preferences['destination']}
Starting Location: {preferences['starting_location']}
Dates: {preferences['departure_date']} to {preferences['return_date']}
Travelers: {preferences['num_travelers']} ({preferences['adults']} adults, {preferences['children']} children, {preferences['seniors']} seniors)
Transportation: {preferences['transportation']}
Accommodation Preferences: {', '.join(preferences['accommodation']) if preferences['accommodation'] else 'Standard options'}
Interests: {', '.join(preferences['interests']) if preferences['interests'] else 'General sightseeing'}
Trip Type: {preferences['trip_type']}
Currency: {currency} ({currency_symbol})
{f"Dietary Preferences: {preferences['dietary_preferences']}" if preferences.get('dietary_preferences') else ""}
{f"Accessibility: {preferences['accessibility_requirements']}" if preferences.get('accessibility_requirements') else ""}

{tier_instructions}

Create a complete {plan_type} plan with:
1. Day-wise detailed itinerary (activities, timings, locations)
2. Transportation details (mention exact mode: train, bus, flight, private car)
3. Accommodation recommendations (name + type + cost per night)
4. Restaurant suggestions for each day, matching the {plan_type} tier
5. Estimated costs breakdown - ALL VALUES IN {currency}
6. Cumulative trip expenditure up to each day
7. Distinguish FIXED costs (travel, stay) from VARIABLE costs (food, activities)

Return ONLY valid JSON, no markdown formatting, no code fences, no explanatory text before or after. Match this exact structure:
{{
  "plan_type": "{plan_type}",
  "currency": "{currency}",
  "itinerary": {{
    "day_1": {{
      "date": "YYYY-MM-DD",
      "activities": [
        {{"time": "09:00", "activity": "description", "location": "place name", "cost": 0, "category": "free/sightseeing/transport"}}
      ],
      "accommodation": {{"name": "specific hotel/hostel name", "type": "hostel/guesthouse/homestay/hotel/resort", "cost": 800, "location": "area"}},
      "meals": [
        {{"time": "breakfast", "restaurant": "name or local eatery", "cuisine": "type", "cost": 150}}
      ],
      "transportation": {{"mode": "train/bus/flight/car", "details": "route info", "cost": 1200}},
      "daily_total": 2500,
      "cumulative_total": 2500,
      "fixed_costs": 2000,
      "variable_costs": 500
    }}
  }},
  "cost_breakdown": {{
    "transportation": 0,
    "accommodation": 0,
    "food": 0,
    "activities": 0,
    "miscellaneous": 0
  }},
  "total_cost": 0,
  "currency_symbol": "{currency_symbol}",
  "highlights": ["highlight 1", "highlight 2", "highlight 3"],
  "budget_tips": ["tip 1", "tip 2", "tip 3"]
}}

Provide REALISTIC Indian market prices in {currency} appropriate for the {plan_type} tier. Be specific with real-sounding restaurant/hotel names. Generate one day entry for each day of the trip duration."""
    
    import json
    
    async def _attempt_generation() -> Dict:
        """Single attempt at calling the AI and parsing its response"""
        chat = LlmChat(
            api_key=EMERGENT_LLM_KEY,
            session_id=f"{trip_id}_{plan_type}_{uuid.uuid4().hex[:6]}",
            system_message=f"You are an expert Indian travel planner. Always respond with ONLY valid JSON (no markdown, no code fences, no extra text). All costs must be in {currency}."
        ).with_model("openai", "gpt-4o")
        
        full_response = ""
        async for event in chat.stream_message(UserMessage(text=prompt)):
            if isinstance(event, TextDelta):
                full_response += event.content
            elif isinstance(event, StreamDone):
                break
        
        # Strip markdown code fences if the model added them despite instructions
        cleaned = full_response.strip()
        if cleaned.startswith("```"):
            cleaned = cleaned.split("```")[1]
            if cleaned.lower().startswith("json"):
                cleaned = cleaned[4:]
        
        start_idx = cleaned.find('{')
        end_idx = cleaned.rfind('}') + 1
        if start_idx == -1 or end_idx <= start_idx:
            raise ValueError("No JSON object found in AI response")
        
        json_str = cleaned[start_idx:end_idx]
        return json.loads(json_str)
    
    last_error = None
    for attempt in range(2):  # try once, retry once on failure
        try:
            plan_data = await _attempt_generation()
            plan_data.setdefault('currency', currency)
            plan_data.setdefault('currency_symbol', currency_symbol)
            plan_data.setdefault('plan_type', plan_type)
            return plan_data
        except Exception as e:
            last_error = e
            logger.warning(f"Attempt {attempt + 1} failed generating {plan_type} plan for trip {trip_id}: {e}")
    
    # Both attempts failed — return a clearly-marked fallback instead of a silent empty plan
    logger.error(f"All attempts failed generating {plan_type} plan: {last_error}")
    return {
        "plan_type": plan_type,
        "currency": currency,
        "currency_symbol": currency_symbol,
        "itinerary": {},
        "cost_breakdown": {
            "transportation": 0,
            "accommodation": 0,
            "food": 0,
            "activities": 0,
            "miscellaneous": 0
        },
        "total_cost": 0,
        "highlights": [],
        "budget_tips": [],
        "error": "We couldn't generate this plan right now. Please try again.",
        "_debug_error": str(last_error)
    }
